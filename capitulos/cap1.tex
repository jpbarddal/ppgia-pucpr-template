No Algoritmo \ref{alg:cap} temos um exemplo de um algoritmo.

\begin{algorithm}
\caption{Um algoritmo}\label{alg:cap}
\begin{algorithmic}[1]
\Require $n \geq 0$
\Ensure $y = x^n$
\State $y \gets 1$
\State $X \gets x$
\State $N \gets n$
\While{$N \neq 0$}
\If{$N$ is even}
    \State $X \gets X \times X$
    \State $N \gets \frac{N}{2}$  \Comment{Um comentário}
\ElsIf{$N$ is odd}
    \State $y \gets y \times X$
    \State $N \gets N - 1$
\EndIf
\EndWhile
\end{algorithmic}
\end{algorithm}

	def fibonacci(n):
		if n <= 1:
			return n
		else:
			return fibonacci(n-1) + fibonacci(n-2)
